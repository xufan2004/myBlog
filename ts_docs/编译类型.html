# 编译类型(配置文件)

在项目根目录中新建tsconfig.json文件可对ts进行配置，ts会根据配置项进行编译


{
    // 编译器的选项，决定了编译器的执行
    "compilerOptions": {

        // 用来指定ts被编译出来js的es的版本
        // "target":"ESNext" 指定最新版的es
        // 可用值：es3,es5,es2015,es2016,es2017,es2018,es2019,es2020，ESNext
        "target":"es2016",

        // 指定要模块化的规范
        // 可用值：commonjs,none,amd,system,umd,es6,es2015,es2020,esnext
        "module":"commonjs",

        // 定制项目中要使用的库,一般不需要设置
        // 可用值： 'es5','es6'，'es2015 ，es7 ，'es2016 ， es2017 ，'es2018','es2019', 'es2020', 'enext', "dom', 'dom.iterable', 'webworker', "webworker.importscripts', 'webworker,iterable', 'scripthost''es2015.core', 'es2015.collection', 'es2015.generator', 'es2015.iterable', 'es2015,promise', 'es2015.proxy','es2015.reflect','es2015.symbol','es2015.symbol.wellknown', 'es2016,array.include', 'es2017object', 'es2017sharedmemory', "es2017.string','es2017.intl',"es2017.typedarrays', "es2018.asyncgenerator', "es2018.asynciterable', "es2018.intl', 'es2018.promise'es2018.regexp','es2019.array',"es2019.object', 'es2019 string","es2019 symbol',"es2020bigint',"es2020,promise', "es2020.sharedmemory''es2020.string', 'es2020.symbol.wellknown', 'es220.intl', 'esnext.array', 'esnext.symbol', 'esnext.asynciterable'esnext.intl', 'esnext.bigint', 'esnext.string', 'esnext.promise', 'esnext.weakref'
        "lib":["dom","ES6"],
        
        // 用来指定编译后文件所在的目录
        "outDir":"", // 例:./dist 路径

        // 将代码合并为一个文件
        "outFile":"", // 例:./dist/app.js 路径

        // 是否对js文件进行编译，默认false
        "allowJs":false,

        // 检查js代码是否符合语法，默认false
        "checkJs": false,

        // 是否移除注释
        "removeComments": true, 

        // 不生成编译后的文件，默认false
        "noEmit": false,

        // 当代码出现错误时，不生成编译后的文件，默认false
        "noEmitOnError": true, 
        
        // 用来设置编译后的文件是否使用严格模式，默认true
        "alwaysScript":false,

        // 不允许隐式的any类型
        "noImplicitAny": true,

        // 不允许类型不明确的this
        "noImplicitThis": true,

        // 严格的检查空值，不允许可能存在的空值
        "strictNullChecks":true,
        
        // 所有严格检查的开关
        // 相当于启用noImplicitAny, --noImplicitThis, --alwaysStrict， --strictNullChecks和 --strictFunctionTypes和--strictPropertyInitialization
        "strict":true

        // ...................
    },
    // 表示编译根目录下所有文件下下的所有ts文件 ** 表示任意目录 * 表示任意文件
    // include 指定要编译的文件
    "include":[
        './scr/**/*'
    ],
    // include 指定不要编译的文件
    "exclude":[
        ''  // 不需要被编译文件的路径
        /* 默认值 "node_modules","bower_components",
        "jspm_packages"
        */
    ],
    // 定义继承的配置文件，可以引用别人定义好的配置项
    "extends": '',//路径
    // 只要想要编译的文件
    "files":[
        '' // 文件名
    ]
}

也可以使用tsc --init初始化后在进行配置
