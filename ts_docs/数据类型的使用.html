# 数据类型的使用

# 显式类型

//number
let a:number // a被设置成了number类型，往后赋值只能为数字
a = 1 // 正确赋值
a = "1" || true || ..... // TS编译器会报错

//string
let a:string //a被设置成了string类型，往后赋值只能为字符串
a = "hello" // 正确赋值
a = 1 || true || ..... // TS编译器会报错

//boolean
let a:boolean //a被设置成了boolean类型，往后赋值只能为布尔值
a = true || false // 正确赋值
a = 1 || "1" || ..... // TS编译器会报错

//字面量
let a:1 //a被设置成了字面量1，往后a赋值只能为1
a = 1 // 正确赋值
a = ..... // 任何值TS编译器都会报错

// 也可以这样写定义多个
let a:number | string
此时a的可以被设置成字符串或数字

// any
let a:any
a = 1 || true || "1" || ....
// 设置any可以赋值任意类型的值，相当于关闭了TS的类型校验

// unknown
let a:unknown // 设置unknown可以赋值任意类型的值，类似于any

// unknown与any的区别
let a:any
let b:unknown
let c:string

c = a // any可以给任意类型赋值，但是会影响被赋值变量的类型 let c:string 变成了 c:any

c = b // TS编译器会报错，不能直接赋值给string类型

// 需要这样赋值(类型校验) OR 类型断言

if(typeof b === 'string' ){
    <!-- 类型校验 -->
    c = b // TS编译器不会报错，正常赋值
}

c = b as string // 类型断言 可以用来告诉解析器变量的实际类型 类似于 c = <string>b
注意：不可 c = b<string>

// void
function fn():void{
    return 1 || ......  //  编译器报错
    return undefined // ok
    return // ok
}

// never
function fn():never{
    return ......  //  编译器报错 不能有任何返回值 不常用
}

// object
let a:object //有点鸡肋 很少使用常用来限制对象里的值，而不是限制类型

let b:{ name:string } // 对象里的值名为name且值必须是字符串
b = {
    name:"333"
}

let d:{ name:string, [proName:string]:any}
// 表示对象中除name外可以有任意类型的键值

let e:{ name:string, age?:number } // 表示对象中必须有一个name属性且值为string，age可填可不填

// array

// 类型[] || Array<类型>

let a:string[] // a被设置成了数组类型，且往后赋值只能为string

let a:number[] // 意义同上

let a:Array<number> // 同 a:number 一样数组中只能包含数字

// tuple(固定长度的数组)
[类型,类型]

let a:[string,number] // a的类型为数组，且a中的内容第一个只能是string，第二个是number
且不存在第三个

// enum(枚举)

enum a {
    b = 0,
    c = "你好"
}

用法 let d = a.b || a.c

// 函数的使用

let a:(a:number,b:number) => number
//定义一个方法 参数都为number ，返回值也会number

a = function(a,b){
    return 值 // number
}

function sum(a:number,b:number){
    return a+b
}

sum(123,456) // return 579
sum(123,"456") // TS编译器会报错

// 定义方法的返回值类型为number
function sum(a,b):number{
    return a+b
}

sum("hello","word") // return "helloword" TS编译器会报错

# 隐式类型

let a;
// TS编译器会把a的类型设置为any let a:any

let a = 1
// TS编译器会把a的类型设置为number let a:number

let a = "1"
// TS编译器会把a的类型设置为string let a:string

let a = true || false
// TS编译器会把a的类型设置为boolean let a:boolean

# 类型的别名

let a: 1 | 2 | 3 | 5
let b: 1 | 2 | 3 | 5

// 上发过于麻烦且不清晰，定义一个类型别名
// 语法

type 别名 = 类型

type numList = 1 | 2 | 3 | 5

let a:numList == let a: 1 | 2 | 3 | 5
